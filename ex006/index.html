Python

Função fatorial desenvolvida em python:

def fatorial(n):
    if n == 0:
        return 1
    else:
        return n * fatorial(n-1)

C

Exemplo de código em C:

#include <stdio.h>

int main() {
    int i, n, fatorial = 1;

    printf("Digite um número: ");
    scanf("%d", &n);

    for (i = 1; i <= n; i++) {
        fatorial = fatorial * i;
    }

    printf("Fatorial de %d = %d", n, fatorial);
    return 0;
}
